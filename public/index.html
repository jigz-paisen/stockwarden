MySQL Schema
Database structure reference
CREATE DATABASE stockwarden;
USE stockwarden;

CREATE TABLE categories (
  id          INT UNSIGNED    NOT NULL AUTO_INCREMENT,
  name        VARCHAR(100)    NOT NULL UNIQUE,
  description TEXT,
  created_at  TIMESTAMP       DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY (id)
);

CREATE TABLE products (
  id            INT UNSIGNED    NOT NULL AUTO_INCREMENT,
  sku           VARCHAR(50)     NOT NULL UNIQUE,
  name          VARCHAR(255)    NOT NULL,
  category_id   INT UNSIGNED,
  quantity      INT             NOT NULL DEFAULT 0,
  reorder_point INT             NOT NULL DEFAULT 10,
  price         DECIMAL(10,2)   NOT NULL DEFAULT 0.00,
  cost          DECIMAL(10,2)   NOT NULL DEFAULT 0.00,
  supplier      VARCHAR(255),
  location      VARCHAR(100),
  created_at    TIMESTAMP       NOT NULL DEFAULT CURRENT_TIMESTAMP,
  updated_at    TIMESTAMP       NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  PRIMARY KEY (id),
  FOREIGN KEY (category_id) REFERENCES categories(id) ON DELETE SET NULL
);

CREATE TABLE transactions (
  id         INT UNSIGNED  NOT NULL AUTO_INCREMENT,
  product_id INT UNSIGNED  NOT NULL,
  type       ENUM('stock_in','stock_out','adjustment','return') NOT NULL,
  quantity   INT           NOT NULL,
  note       VARCHAR(500),
  created_by VARCHAR(100),
  created_at TIMESTAMP     NOT NULL DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY (id),
  FOREIGN KEY (product_id) REFERENCES products(id) ON DELETE CASCADE
);

CREATE TABLE users (
  id         INT UNSIGNED  NOT NULL AUTO_INCREMENT,
  username   VARCHAR(80)   NOT NULL UNIQUE,
  email      VARCHAR(255)  NOT NULL UNIQUE,
  password   VARCHAR(255)  NOT NULL,
  role       ENUM('admin','manager','viewer') NOT NULL DEFAULT 'viewer',
  is_active  TINYINT(1)    NOT NULL DEFAULT 1,
  created_at TIMESTAMP     NOT NULL DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY (id)
);

CREATE VIEW v_inventory_status AS
SELECT p.*, c.name AS category_name,
  (p.quantity * p.cost) AS stock_value,
  CASE WHEN p.quantity = 0 THEN 'out_of_stock'
       WHEN p.quantity <= p.reorder_point THEN 'low_stock'
       ELSE 'in_stock' END AS stock_status
FROM products p
LEFT JOIN categories c ON p.category_id = c.id;

CREATE VIEW v_transactions AS
SELECT t.*, p.name AS product_name, p.sku AS product_sku
FROM transactions t
JOIN products p ON t.product_id = p.id;
