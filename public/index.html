📁 Tables & Views
-- ================================================ -- StockWarden Inventory Management MySQL Schema -- ================================================ CREATE DATABASE IF NOT EXISTS stockwarden CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci; USE stockwarden; -- Categories table CREATE TABLE categories ( id INT UNSIGNED AUTO_INCREMENT PRIMARY KEY, name VARCHAR(100) NOT NULL UNIQUE, description TEXT, created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ); -- Products / Inventory table CREATE TABLE products ( id INT UNSIGNED AUTO_INCREMENT PRIMARY KEY, sku VARCHAR(50) NOT NULL UNIQUE, name VARCHAR(255) NOT NULL, description TEXT, category_id INT UNSIGNED, quantity INT NOT NULL DEFAULT 0, reorder_point INT NOT NULL DEFAULT 10, price DECIMAL(10,2) NOT NULL DEFAULT 0.00, cost DECIMAL(10,2) NOT NULL DEFAULT 0.00, supplier VARCHAR(255), location VARCHAR(100), created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP, updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP, FOREIGN KEY (category_id) REFERENCES categories(id) ON DELETE SET NULL, INDEX idx_sku (sku), INDEX idx_category (category_id), INDEX idx_quantity (quantity) ); -- Stock Transactions table CREATE TABLE transactions ( id INT UNSIGNED AUTO_INCREMENT PRIMARY KEY, product_id INT UNSIGNED NOT NULL, type ENUM('stock_in', 'stock_out', 'adjustment', 'return') NOT NULL, quantity INT NOT NULL, note VARCHAR(500), created_by VARCHAR(100), created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP, FOREIGN KEY (product_id) REFERENCES products(id) ON DELETE CASCADE, INDEX idx_product (product_id), INDEX idx_type (type), INDEX idx_date (created_at) ); -- Users table (for multi-user access) CREATE TABLE users ( id INT UNSIGNED AUTO_INCREMENT PRIMARY KEY, username VARCHAR(80) NOT NULL UNIQUE, email VARCHAR(255) NOT NULL UNIQUE, password VARCHAR(255) NOT NULL, -- bcrypt hash role ENUM('admin', 'manager', 'viewer') DEFAULT 'viewer', created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ); -- ================================================ -- Useful Views -- ================================================ CREATE VIEW v_inventory_status AS SELECT p.*, c.name AS category_name, (p.quantity * p.cost) AS stock_value, CASE WHEN p.quantity = 0 THEN 'out_of_stock' WHEN p.quantity <= p.reorder_point THEN 'low_stock' ELSE 'in_stock' END AS stock_status FROM products p LEFT JOIN categories c ON p.category_id = c.id; CREATE VIEW v_transactions AS SELECT t.*, p.name AS product_name, p.sku AS product_sku FROM transactions t JOIN products p ON t.product_id = p.id; -- ================================================ -- Stored Procedure: Stock Adjustment -- ================================================ DELIMITER // CREATE PROCEDURE adjust_stock( IN p_product_id INT, IN p_qty_change INT, IN p_type VARCHAR(20), IN p_note VARCHAR(500) ) BEGIN UPDATE products SET quantity = quantity + p_qty_change WHERE id = p_product_id; INSERT INTO transactions (product_id, type, quantity, note) VALUES (p_product_id, p_type, p_qty_change, p_note); END // DELIMITER ;
